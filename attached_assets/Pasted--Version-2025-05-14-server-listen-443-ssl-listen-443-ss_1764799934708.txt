server {
    listen 443 ssl;
    listen [::]:443 ssl;

    server_name stash.*;

    include /config/nginx/ssl.conf;

    if ($lan-ip = yes) { set $geo-whitelist yes; }
    if ($geo-whitelist = no) { return 404; }
    client_max_body_size 100m;

    location / {
        include /config/nginx/proxy-confs/organizr-auth.subfolder.conf;
#        auth_request /auth-0;
        include /config/nginx/proxy.conf;
        include /config/nginx/resolver.conf;

        set $upstream_app stash;
        set $upstream_port 9999;
        set $upstream_proto https;
        proxy_pass $upstream_proto://$upstream_app:$upstream_port;

        access_log /config/log/nginx/stash_access.log;
        error_log /config/log/nginx/stash_error.log;
        proxy_ssl_server_name on; proxy_ssl_name stash.feldorn.com;
   }

    # Authenticated access for /graphql (all IPs)
    location ^~ /graphql {
        include /config/nginx/proxy-confs/organizr-auth.subfolder.conf;
#        auth_request /auth-0;
        include /config/nginx/proxy.conf;
        include /config/nginx/resolver.conf;

        error_page 401 = $scheme://$host/;

        set $upstream_app stash;
        set $upstream_port 9999;
        set $upstream_proto https;
        proxy_pass $upstream_proto://$upstream_app:$upstream_port/graphql;

        access_log /config/log/nginx/stash_graphql_access.log;
        error_log /config/log/nginx/stash_graphql_error.log;
    }

    # Unauthenticated access for /graphql-local (192.168.0.200 only)
    location ^~ /graphql-local {
        allow 192.168.0.200;
        allow 127.0.0.1;
        deny all;

        include /config/nginx/proxy.conf;
        include /config/nginx/resolver.conf;

        set $upstream_app stash;
        set $upstream_port 9999;
        set $upstream_proto https;
        proxy_pass $upstream_proto://$upstream_app:$upstream_port/graphql;

        access_log /config/log/nginx/stash_graphql_local_access.log;
        error_log /config/log/nginx/stash_graphql_local_error.log;
    }
}
